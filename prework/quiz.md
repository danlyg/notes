**20) How might you write a function called `maxNumber` that gives us the biggest value from a set of four numbers?**

Answer:
```
function maxNumber(a, b, c, d) {
	array = [a, b, c, d];
	array.sort(function(a, b){
	        return a - b;
	})
	return array[array.length-1];
}
```

or

```
function maxNumber(a, b, c, d) {
	if (a >= b && a >= c && a >= d) {
		 return a;
	} else if (b >= a && b >= c && b >= d) {
		return b;
	} else if (c >= a && c >= b && c >= d) {
		return c;
	} else {
		return d;
	}
}
```
